24// Copy from cbor4ii/core/dec.rs.
25#[inline]
26pub(crate) fn peek_one<'a, R: dec::Read<'a>>(
27    name: &'static str,
28    reader: &mut R,
29) -> Result<u8, DecodeError<R::Error>> {
30    let buf = match reader.fill(1)? {
31        dec::Reference::Long(buf) => buf,
32        dec::Reference::Short(buf) => buf,
33    };
34    let byte = buf.first().copied().ok_or(DecodeError::Eof {
35        name,
36        expect: Len::new(1),
37    })?;
38    Ok(byte)
39}

41// Copy from cbor4ii/core/dec.rs.
42#[inline]
43pub(crate) fn pull_one<'a, R: dec::Read<'a>>(
44    name: &'static str,
45    reader: &mut R,
46) -> Result<u8, DecodeError<R::Error>> {
47    let byte = peek_one(name, reader)?;
48    reader.advance(1);
49    Ok(byte)
